export interface DashboardStats {
  monthlySpend: number;         // sum of cost_usd for current calendar month
  activeProjects: number;       // count of projects with status='active'
  connectionCount: number;      // total api_connections count (any status)
  alertCount: number;           // count of alert_log rows fired this calendar month
}

export interface SpendRow {
  date: string;                 // "YYYY-MM-DD"
  openai: number;
  anthropic: number;
  replicate: number;
  falai: number;
  gemini: number;
  groq: number;
  mistral: number;
  [key: string]: number | string;
}

export interface TopProject {
  id: string;
  name: string;
  monthlySpend: number;
  budgetLimit: number | null;   // from budget_rules, null if no rule
}

export interface RecentAlert {
  id: string;
  firedAt: string;              // ISO timestamp
  message: string;              // computed — see below
  type: string;                 // computed from action_taken
  severity: "critical" | "warning" | "info";  // computed from percent_used
  status: "active" | "resolved" | "acknowledged";
}
```

Implement `getDashboardStats(supabase, userId)`:
- Monthly spend: query `usage_records` WHERE `user_id = userId` AND `date >= first day of current UTC month` AND `date <= today`. Sum `cost_usd`. Use `.gte("date", calMonthStart).lte("date", today)`.
- Active projects: query `projects` WHERE `user_id = userId` AND `status = 'active'`. Use `.select("id", { count: "exact", head: true })`.
- Connection count: query `api_connections` WHERE `user_id = userId`. Count with `{ count: "exact", head: true }`.
- Alert count: query `alert_log` WHERE `user_id = userId` AND `fired_at >= calMonthStart`. Count with `{ count: "exact", head: true }`.
- Return `DashboardStats`.

Implement `getDailySpend(supabase, userId, days: number)`:
- `days` is 7, 30, or 90.
- Compute `startDate` as `days` days ago from today (UTC, date string).
- Query `usage_records` joined to `api_connections` for the `provider` field:
  ```
  supabase
    .from("usage_records")
    .select("date, cost_usd, api_connections!inner(provider)")
    .eq("user_id", userId)
    .gte("date", startDate)
    .order("date", { ascending: true })
  ```
- Pivot in TypeScript: group by date, accumulate `cost_usd` per provider. Provider column name from DB is the `provider` field value as stored (e.g. 'openai', 'anthropic', 'replicate', 'falai', 'gemini', 'groq', 'mistral'). Map directly to SpendRow fields.
- Build a complete date skeleton from `startDate` to today (no gaps), filling zero for missing dates.
- Return `SpendRow[]` sorted by date ascending.

Implement `getTopProjects(supabase, userId)`:
- Query the `project_monthly_spend` view (safe to use — RLS via invoker rights): `supabase.from("project_monthly_spend").select("id, name, monthly_spend").eq("user_id", userId).order("monthly_spend", { ascending: false }).limit(3)`.
- For each project, attempt to find its lowest limit_usd from `budget_rules` WHERE `project_id = project.id`. Do this with a single `in` query across the 3 project ids, not 3 separate queries.
- Return `TopProject[]`.

export interface ProjectStats {
  id: string;
  name: string;
  description: string | null;
  status: string;
  monthlySpend: number;
  burnRateDaily: number;        // avg daily spend over last 7 days (sum of last 7 days / 7)

  budgetLimit: number | null;
  lastUpdatedAt: string | null; // MAX(last_polled_at) from active connections
}

export interface ProjectConnection {
  id: string;
  provider: string;
  apiKeySuffix: string;         // api_key_suffix column
  status: string;
  lastPolledAt: string | null;  // ISO timestamp from last_polled_at column
}

export interface ProjectAlert {
  id: string;
  firedAt: string;
  severity: "critical" | "warning" | "info";
  message: string;
  status: "active" | "resolved" | "acknowledged";
}
```

Implement `getRecentAlerts(supabase, userId, limit = 5)`:
- Query `alert_log` WHERE `user_id = userId` ORDER BY `fired_at` DESC LIMIT `limit`.
- Select columns: `id, fired_at, action_taken, spend_at_trigger, limit_usd, percent_used, status`.
- Compute `severity` from `percent_used`: >= 100 → "critical", >= 80 → "warning", else → "info".
- Compute `type` from `action_taken`: "block" → "Budget Limit", anything else → "Budget Threshold".
- Compute `message`: `"Spend $${spend_at_trigger.toFixed(2)} reached ${percent_used}% of $${limit_usd} ${type} rule"`.
- Omit provider (not available without a multi-hop join, per research).
- Return `RecentAlert[]`.

Implement `getProjectStats(supabase, userId, projectId: string)`:
- Fetch project row: `projects` WHERE `id = projectId AND user_id = userId`. If not found, return null.
- Monthly spend: sum `cost_usd` from `usage_records` WHERE `user_id = userId AND project_id IS NULL` is wrong — `usage_records` has no `project_id`. Instead join via `api_connections`: sum `cost_usd` from `usage_records` WHERE `connection_id IN (SELECT id FROM api_connections WHERE project_id = projectId AND user_id = userId)` AND `date >= calMonthStart`. Use `.eq("user_id", userId).gte("date", calMonthStart)` then filter by connection_id in JS from a pre-fetched connections list.
- Burn rate: same but last 7 days, divide by 7.
- Budget limit: lowest `limit_usd` from `budget_rules` WHERE `project_id = projectId`.
- Last updated: MAX `last_polled_at` from `api_connections` WHERE `project_id = projectId AND status = 'active'`.
- Return `ProjectStats | null`.

Implement `getProjectConnections(supabase, userId, projectId: string)`:
- Query `api_connections` WHERE `project_id = projectId AND user_id = userId`.
- Select `id, provider, api_key_suffix, status, last_polled_at`.
- Return `ProjectConnection[]`.

Implement `getProjectAlerts(supabase, userId, projectId: string, limit = 20)`:
- `alert_log` has a `project_id` column. Query WHERE `project_id = projectId AND user_id = userId` ORDER BY `fired_at` DESC LIMIT `limit`.
- Apply same severity/message/type computation as `getRecentAlerts`.
- Return `ProjectAlert[]`.
